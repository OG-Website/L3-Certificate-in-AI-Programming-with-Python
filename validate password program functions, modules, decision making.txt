import re  # imports the built-in regular expression module


def has_minimum_length(password):
    # Checks that the password is at least 8 characters long
    return len(password) >= 8


def has_lowercase_letter(password):
    # Checks that the password contains at least one lowercase letter
    return re.search("[a-z]", password) is not None


def has_uppercase_letter(password):
    # Checks that the password contains at least one uppercase letter
    return re.search("[A-Z]", password) is not None


def has_number(password):
    # Checks that the password contains at least one number
    return re.search("[0-9]", password) is not None


def validate_password(password):
    # Uses helper functions to check all password rules
    return (
        has_minimum_length(password)
        and has_lowercase_letter(password)
        and has_uppercase_letter(password)
        and has_number(password)
    )


password = input("Create a new password: ")

if validate_password(password):
    print("Valid password")
else:
    print("Invalid password, try again.")
