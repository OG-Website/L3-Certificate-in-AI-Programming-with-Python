Practical Code Used
-------------------

```python
# OG Web.site Unit 3 - Statistical Analysis practical evidence
# This notebook uses the local Titanic CSV file from the active Unit 3 folder.

from pathlib import Path
import pandas as pd

csv_path = Path("titanic.csv")
if not csv_path.exists():
    csv_path = Path("data") / "titanic.csv"

# Load the CSV into a DataFrame so the data can be inspected and analysed.
titanic_df = pd.read_csv(csv_path)

print("Rows and columns:", titanic_df.shape)
print("First five rows:")
print(titanic_df.head())

print("Data types:")
print(titanic_df.dtypes)

print("Summary statistics:")
print(titanic_df.describe())

numeric_columns = ["Age", "Fare", "SibSp", "Parch", "Survived"]
print("Mean values:")
print(titanic_df[numeric_columns].mean(numeric_only=True))

print("Median values:")
print(titanic_df[numeric_columns].median(numeric_only=True))

print("Mode values:")
print(titanic_df[numeric_columns].mode().head(1))

print("Standard deviation values:")
print(titanic_df[numeric_columns].std(numeric_only=True))

print("Survival rate by sex:")
print(titanic_df.groupby("Sex")["Survived"].mean())

print("Survival rate by passenger class:")
print(titanic_df.groupby("Pclass")["Survived"].mean())

print("Missing values by column:")
print(titanic_df.isna().sum())
```
